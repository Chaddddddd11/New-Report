function registerInstructor() { const name = document.getElementById('name').value.trim(); const subject = document.getElementById('subject').value.trim() || 'N/A'; const course = courseSelect.value; const year = yearSelect.value; const section = sectionSelect.value; const day = document.getElementById('day').value; const time = document.getElementById('time').value; // Validate all required fields if (!name || !course || !year || !section || !day || !time) { showError("Please fill in all required fields."); return; } // Retrieve instructors from localStorage each time let instructors = JSON.parse(localStorage.getItem('instructors')) || []; // Check for duplicate instructor const isDuplicate = instructors.some(instructor => instructor.name === name && instructor.course === course && instructor.year === year && instructor.section === section ); if (isDuplicate) { showError("This instructor is already registered for this course, year, and section."); return; } const instructor = { name, subject, course, year, section, day, time }; instructors.push(instructor); localStorage.setItem('instructors', JSON.stringify(instructors)); // Show success message alert('Instructor enrolled successfully!'); clearForm(); clearError(); } // Remove the global instructors variable // let instructors = JSON.parse(localStorage.getItem('instructors')) || [];
